// Başlangıç değerleri
let scale = 1;
let offsetX = 0;
let offsetY = 0;
let isDragging = false;
let startX, startY;

const container = document.getElementById('container');
const viewport = document.getElementById('viewport');
const splash = document.getElementById('splash');
const main = document.getElementById('main');
const enterBtn = document.getElementById('enterBtn');

// Giriş butonu
enterBtn.addEventListener('click', () => {
  splash.classList.add('hidden');
  setTimeout(() => { splash.style.display = 'none'; }, 1000);
  main.classList.remove('hidden');
});

// Zoom fonksiyonu
function zoom(delta, mouseX = window.innerWidth/2, mouseY = window.innerHeight/2) {
  const oldScale = scale;
  scale = Math.max(0.3, Math.min(8, scale + delta)); // min 0.3x max 8x

  // Fare pozisyonuna göre merkezleme
  const rect = viewport.getBoundingClientRect();
  const mx = mouseX - rect.left;
  const my = mouseY - rect.top;

  offsetX = mx - (mx - offsetX) * (scale / oldScale);
  offsetY = my - (my - offsetY) * (scale / oldScale);

  updateTransform();
}

function updateTransform() {
  container.style.transform = `translate(${offsetX}px, ${offsetY}px) scale(${scale})`;
}

// Mouse tekerleği ile zoom
viewport.addEventListener('wheel', e => {
  e.preventDefault();
  const delta = e.deltaY < 0 ? 0.15 : -0.15;
  zoom(delta, e.clientX, e.clientY);
});

// Dokunmatik pinch zoom (basit versiyon)
let prevDist = 0;
viewport.addEventListener('touchmove', e => {
  if (e.touches.length === 2) {
    e.preventDefault();
    const dx = e.touches[0].pageX - e.touches[1].pageX;
    const dy = e.touches[0].pageY - e.touches[1].pageY;
    const dist = Math.hypot(dx, dy);

    if (prevDist > 0) {
      const delta = (dist - prevDist) / 800;
      zoom(delta, (e.touches[0].pageX + e.touches[1].pageX)/2, (e.touches[0].pageY + e.touches[1].pageY)/2);
    }
    prevDist = dist;
  }
}, { passive: false });

viewport.addEventListener('touchend', () => { prevDist = 0; });

// Sürükleme (pan)
viewport.addEventListener('mousedown', e => {
  isDragging = true;
  startX = e.clientX - offsetX;
  startY = e.clientY - offsetY;
  viewport.style.cursor = 'grabbing';
});

viewport.addEventListener('mousemove', e => {
  if (!isDragging) return;
  offsetX = e.clientX - startX;
  offsetY = e.clientY - startY;
  updateTransform();
});

viewport.addEventListener('mouseup mouseleave', () => {
  isDragging = false;
  viewport.style.cursor = 'grab';
});

// Butonlar
document.getElementById('zoomIn').onclick = () => zoom(0.3);
document.getElementById('zoomOut').onclick = () => zoom(-0.3);
document.getElementById('reset').onclick = () => {
  scale = 1; offsetX = 0; offsetY = 0;
  updateTransform();
};

// İlk konumlandırma
updateTransform();
